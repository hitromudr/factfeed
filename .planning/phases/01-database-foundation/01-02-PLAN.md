Step 3 — Rewrite alembic/env.py with full async configuration AND the GIN index autogenerate exclusion workaround. The complete env.py must be:

```python
import asyncio
import os
from logging.config import fileConfig

from sqlalchemy import pool
from sqlalchemy.engine import Connection
from sqlalchemy.ext.asyncio import async_engine_from_config

from alembic import context

# Import Base metadata so Alembic knows about our models
from factfeed.db.models import Base

# Alembic Config object — access alembic.ini values
config = context.config

# Set up Python logging from alembic.ini [loggers] section
if config.config_file_name is not None:
    fileConfig(config.config_file_name)

# Allow DATABASE_URL env var to override alembic.ini sqlalchemy.url
database_url = os.environ.get("DATABASE_URL")
if database_url:
    config.set_main_option("sqlalchemy.url", database_url)

target_metadata = Base.metadata


def do_run_migrations(connection: Connection) -> None:
    context.configure(
        connection=connection,
        target_metadata=target_metadata,
        # CRITICAL: Exclude the GIN index from autogenerate comparison.
        # Alembic false-positively detects the tsvector GIN index as "changed"
        # on every autogenerate run because PostgreSQL returns type casts that
        # Alembic's string comparison does not match.
        # See: https://github.com/sqlalchemy/alembic/issues/1390
        include_name=lambda name, type_, parent_names: (
            name not in {"ix_articles_search_vector"} if type_ == "index" else True
        ),
    )
    with context.begin_transaction():
        context.run_migrations()


async def run_async_migrations() -> None:
    """Run migrations using async engine (required for asyncpg driver)."""
    connectable = async_engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    async with connectable.connect() as connection:
        await connection.run_sync(do_run_migrations)
    await connectable.dispose()


def run_migrations_online() -> None:
    asyncio.run(run_async_migrations())


def run_migrations_offline() -> None:
    """Offline mode: generate SQL script without connecting."""
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )
    with context.begin_transaction():
        context.run_migrations()


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

Step 4 — Create alembic/versions/0001_initial_schema.py (hand-written, NOT autogenerated):

```python
"""Initial schema: sources, articles, sentences

Revision ID: 0001
Revises:
Create Date: 2026-02-23

IMPORTANT: This migration is hand-written — do NOT replace with autogenerated output.
Alembic autogenerate cannot produce GENERATED ALWAYS AS STORED syntax for tsvector
columns, and produces false-positive drift detection on the GIN index. This migration
is the ground truth for the schema.
"""
from alembic import op
import sqlalchemy as sa

revision = "0001"
down_revision = None
branch_labels = None
depends_on = None


def upgrade() -> None:
    # Create sources table first (articles FK references it)
    op.create_table(
        "sources",
        sa.Column("id", sa.Integer(), primary_key=True),
        sa.Column("name", sa.String(100), nullable=False),
        sa.Column("feed_url", sa.Text(), nullable=False, unique=True),
    )

    # Create articles table (without search_vector — added via raw SQL below)
    op.create_table(
        "articles",
        sa.Column("id", sa.Integer(), primary_key=True),
        sa.Column("url", sa.Text(), nullable=False),
        sa.Column("url_hash", sa.String(64), nullable=False),
        sa.Column("title", sa.Text(), nullable=False),
        sa.Column("body", sa.Text(), nullable=True),
        sa.Column("published_at", sa.DateTime(timezone=True), nullable=True),
        sa.Column("source_id", sa.Integer(), sa.ForeignKey("sources.id"), nullable=True),
    )

    # url_hash unique constraint for deduplication (sha256 hex of normalized URL)
    op.create_unique_constraint("uq_articles_url_hash", "articles", ["url_hash"])

    # GENERATED ALWAYS AS STORED column — must be added via raw SQL.
    # SQLAlchemy op.add_column cannot express GENERATED ALWAYS AS syntax.
    # PostgreSQL computes and maintains this column automatically on every INSERT/UPDATE.
    op.execute("""
        ALTER TABLE articles
        ADD COLUMN search_vector tsvector
        GENERATED ALWAYS AS (
            to_tsvector('english', coalesce(title, '') || ' ' || coalesce(body, ''))
        ) STORED
    """)

    # GIN index on the generated tsvector column — enables full-text search.
    # Must be created in migration zero (not retrofitted) to avoid table lock on large tables.
    # This index is excluded from Alembic autogenerate comparison by name in env.py.
    op.execute("""
        CREATE INDEX ix_articles_search_vector ON articles USING GIN (search_vector)
    """)

    # Create sentences table — child of articles, not a JSON column.
    # ondelete CASCADE ensures sentences are deleted when parent article is deleted.
    op.create_table(
        "sentences",
        sa.Column("id", sa.Integer(), primary_key=True),
        sa.Column(
            "article_id",
            sa.Integer(),
            sa.ForeignKey("articles.id", ondelete="CASCADE"),
            nullable=False,
        ),
        sa.Column("position", sa.Integer(), nullable=False),
        sa.Column("text", sa.Text(), nullable=False),
        sa.Column("label", sa.String(20), nullable=True),   # fact/opinion/mixed/unclear
        sa.Column("confidence", sa.Float(), nullable=True),  # 0.0 - 1.0
    )
    # Index on article_id for efficient per-article sentence queries
    op.create_index("ix_sentences_article_id", "sentences", ["article_id"])


def downgrade() -> None:
    op.drop_table("sentences")
    op.drop_index("ix_articles_search_vector", table_name="articles")
    op.drop_table("articles")
    op.drop_table("sources")
```

Do NOT autogenerate the migration. Do NOT use Alembic sync template (only async). Do NOT add the search_vector column via `op.add_column` — use `op.execute` with raw ALTER TABLE SQL only.

Create the Dockerfile and docker-compose.yml. The startup order is critical: postgres must be healthy before migrate runs; migrate must complete successfully before app starts.

Create Dockerfile at the repository root:
```dockerfile
FROM python:3.12-slim

# Install uv
COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /usr/local/bin/

# Set working directory
WORKDIR /app

# Copy dependency files first (layer cache optimization)
COPY pyproject.toml uv.lock* ./

# Install dependencies (no editable install for production)
RUN uv sync --frozen --no-dev

# Copy application code
COPY factfeed/ ./factfeed/
COPY alembic/ ./alembic/
COPY alembic.ini ./

# Default command (overridden in docker-compose.yml per service)
CMD ["uv", "run", "uvicorn", "factfeed.web.main:app", "--host", "0.0.0.0", "--port", "8000"]
```

Note: `uv.lock*` uses a glob so the COPY does not fail if the lockfile does not exist yet. After the first `uv sync`, a lockfile is generated — include it in subsequent builds for reproducibility.

Create docker-compose.yml at the repository root:
```yaml
services:
  postgres:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER: ${POSTGRES_USER:-factfeed}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-factfeed}
      POSTGRES_DB: ${POSTGRES_DB:-factfeed}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER:-factfeed} -d ${POSTGRES_DB:-factfeed}"]
      interval: 5s
      timeout: 5s
      retries: 10
      start_period: 10s

  migrate:
    build: .
    command: ["uv", "run", "alembic", "upgrade", "head"]
    environment:
      DATABASE_URL: postgresql+asyncpg://${POSTGRES_USER:-factfeed}:${POSTGRES_PASSWORD:-factfeed}@postgres:5432/${POSTGRES_DB:-factfeed}
    env_file:
      - path: .env
        required: false
    depends_on:
      postgres:
        condition: service_healthy
    restart: on-failure

  app:
    build: .
    command: ["uv", "run", "uvicorn", "factfeed.web.main:app", "--host", "0.0.0.0", "--port", "8000", "--workers", "1"]
    environment:
      DATABASE_URL: postgresql+asyncpg://${POSTGRES_USER:-factfeed}:${POSTGRES_PASSWORD:-factfeed}@postgres:5432/${POSTGRES_DB:-factfeed}
    env_file:
      - path: .env
        required: false
    ports:
      - "8000:8000"
    depends_on:
      postgres:
        condition: service_healthy
      migrate:
        condition: service_completed_successfully

volumes:
  postgres_data:
```

Important implementation notes:
- Use `postgres:16-alpine` (not latest — pinned for reproducibility)
- DATABASE_URL in each service uses the `postgres` hostname (Docker internal network name), not `localhost`
- `env_file: path: .env, required: false` — graceful when .env absent in CI
- `restart: on-failure` on migrate — retries if postgres wasn't quite ready (belt-and-suspenders beyond health check)
- `--workers 1` on app command — APScheduler single-worker constraint (Phase 2 concern, but set now)
- Do NOT use `depends_on: [postgres]` without a condition — this is the common pitfall that causes connection refused errors
- Do NOT use wait-for-it.sh or any shell entrypoint scripts — use native Compose health conditions
